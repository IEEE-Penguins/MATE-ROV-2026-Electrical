#include <Wire.h>
#include <MPU6050.h>
#include <Kalman.h> // Kalman filter library

// MPU6050 register addresses
#define MPU6050_ADDR 0x68
#define REG_PWR_MGMT_1 0x6B
#define REG_ACCEL_XOUT_H 0x3B
#define REG_TEMP_OUT_H 0x41
#define REG_ACCEL_CONFIG 0x1C
#define REG_GYRO_CONFIG 0x1B

// Thresholds for accident detection
#define ACC_THRESHOLD 2.5   // Acceleration in g
#define GYRO_THRESHOLD 300  // Angular velocity in °/s

// MPU6050 variables
int16_t AcX, AcY, AcZ, GyX, GyY, GyZ, TempRaw;
float accelX, accelY, accelZ, gyroX, gyroY, gyroZ;
float angleX, angleY, temperature, dt = 0.01;

// Kalman filter objects
Kalman kalmanAccX, kalmanAccY, kalmanAccZ;
Kalman kalmanGyroX, kalmanGyroY, kalmanGyroZ;
Kalman kalmanAngleX, kalmanAngleY, kalmanTemp;

void setup() {
  Wire.begin();                 // Start I2C communication
  Serial.begin(115200);         // Start Serial communication

  // Initialize MPU6050
  mpu_init();
}

void loop() {
  mpu_read();            // Read data from MPU6050
  processSensorData();   // Process raw and filtered data
  displayData();         // Display data on Serial Monitor
  delay(10);             // Update data every 10ms
}

// Initialize MPU6050
void mpu_init() {
  Wire.beginTransmission(MPU6050_ADDR);
  Wire.write(REG_PWR_MGMT_1);
  Wire.write(0x00); // Activate MPU6050
  Wire.endTransmission(true);

  Wire.beginTransmission(MPU6050_ADDR);
  Wire.write(REG_ACCEL_CONFIG);
  Wire.write(0x00); // ±2g
  Wire.endTransmission(true);

  Wire.beginTransmission(MPU6050_ADDR);
  Wire.write(REG_GYRO_CONFIG);
  Wire.write(0x00); // ±250°/s
  Wire.endTransmission(true);
  
  Serial.println("MPU6050 initialized successfully.");
}

// Read data from MPU6050
void mpu_read() {
  Wire.beginTransmission(MPU6050_ADDR);
  Wire.write(REG_ACCEL_XOUT_H);
  Wire.endTransmission(false);
  Wire.requestFrom(MPU6050_ADDR, 14, true);

  AcX = Wire.read() << 8 | Wire.read();
  AcY = Wire.read() << 8 | Wire.read();
  AcZ = Wire.read() << 8 | Wire.read();
  TempRaw = Wire.read() << 8 | Wire.read(); // Read temperature
  GyX = Wire.read() << 8 | Wire.read();
  GyY = Wire.read() << 8 | Wire.read();
  GyZ = Wire.read() << 8 | Wire.read();
}

// Process raw and Kalman-filtered data
void processSensorData() {
  // Convert raw accelerometer data to g
  accelX = kalmanAccX.getAngle((float)AcX / 16384.0, 0, dt);
  accelY = kalmanAccY.getAngle((float)AcY / 16384.0, 0, dt);
  accelZ = kalmanAccZ.getAngle((float)AcZ / 16384.0, 0, dt);

  // Convert raw gyroscope data to °/s
  gyroX = kalmanGyroX.getAngle((float)GyX / 131.0, 0, dt);
  gyroY = kalmanGyroY.getAngle((float)GyY / 131.0, 0, dt);
  gyroZ = kalmanGyroZ.getAngle((float)GyZ / 131.0, 0, dt);

  // Calculate angles from accelerometer data
  float accelAngleX = atan(AcY / sqrt(pow(AcX, 2) + pow(AcZ, 2))) * 180 / PI;
  float accelAngleY = atan(-AcX / sqrt(pow(AcY, 2) + pow(AcZ, 2))) * 180 / PI;
  angleX = kalmanAngleX.getAngle(accelAngleX, gyroX, dt);
  angleY = kalmanAngleY.getAngle(accelAngleY, gyroY, dt);

  // Convert raw temperature to degrees Celsius and filter
  float rawTemperature = TempRaw / 340.0 + 36.53;
  temperature = kalmanTemp.getAngle(rawTemperature, 0, dt);
}

// Display processed and raw data
void displayData() {
  Serial.print("Raw Accel (g): X="); Serial.print((float)AcX / 16384.0);
  Serial.print(", Y="); Serial.print((float)AcY / 16384.0);
  Serial.print(", Z="); Serial.println((float)AcZ / 16384.0);

  Serial.print("Raw Gyro (°/s): X="); Serial.print((float)GyX / 131.0);
  Serial.print(", Y="); Serial.print((float)GyY / 131.0);
  Serial.print(", Z="); Serial.println((float)GyZ / 131.0);

  Serial.print("Filtered Accel (g): X="); Serial.print(accelX);
  Serial.print(", Y="); Serial.print(accelY);
  Serial.print(", Z="); Serial.println(accelZ);

  Serial.print("Filtered Gyro (°/s): X="); Serial.print(gyroX);
  Serial.print(", Y="); Serial.print(gyroY);
  Serial.print(", Z="); Serial.println(gyroZ);

  Serial.print("Filtered Angle (°): X="); Serial.print(angleX);
  Serial.print(", Y="); Serial.println(angleY);

  Serial.print("Temperature (°C): "); Serial.println(temperature);
}
